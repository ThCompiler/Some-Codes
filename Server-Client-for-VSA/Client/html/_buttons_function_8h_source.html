60  void descent_down(int vertex, int* swaping, int* comparisons)
61  {
62  assert(swaping != nullptr);
63  assert(comparisons != nullptr);
64  assert(swaping != comparisons);
65 
66  const int child = 2 * vertex;
67  int minIndexOfChild = vertex;
68 
69  if (child < size && tree[child] < tree[minIndexOfChild])
70  minIndexOfChild = child;
71  (*comparisons)++;
72 
73  if (child + 1 < size && tree[child + 1] < tree[minIndexOfChild])
74  minIndexOfChild = child + 1;
75  (*comparisons)++;
76 
77  if (minIndexOfChild != vertex)
78  {
79  std::swap(tree[minIndexOfChild], tree[vertex]);
80  (*swaping)++;
81 
82  descent_down(minIndexOfChild, swaping, comparisons);
83  }

478  void merge(int* swaping, int* comparisons, std::vector<int>* Array, int left, int middle, int right)
479  {
480  assert(swaping != nullptr);
481  assert(comparisons != nullptr);
482  assert(swaping != comparisons);
483  assert(Array != nullptr);
484 
485  int i = middle + 1;
486  int j = left;
487 
488  std::vector<int>merged;

491  {
492  if ((*Array)[i] <= (*Array)[j])
493  merged.push_back((*Array)[i++]);
494  else
495  merged.push_back((*Array)[j++]);
496 
497  (*comparisons)++;
498  (*swaping)++;
499  }
500 
501  while (j <= middle)
502  {
503  merged.push_back((*Array)[j++]);
504  (*swaping)++;
505  }
506 
507  while (i <= right)
508  {
509  merged.push_back((*Array)[i++]);
510  (*swaping)++;
511  }
512 
513  for (i = left; i <= right; i++)
514  {
515  (*Array)[i] = merged[i - left];
516  (*swaping)++;
517  }
