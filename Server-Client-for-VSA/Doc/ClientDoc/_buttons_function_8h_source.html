57  {
58  assert(swaping != nullptr);
59  assert(comparisons != nullptr);
60  assert(swaping != comparisons);
61

64

67  (*comparisons)++;

70  minIndexOfChild = child + 1;
71  (*comparisons)++;
72

74  {
75  std::swap(tree[minIndexOfChild], tree[vertex]);
76  (*swaping)++;
77

461  void merge(int* swaping, int* comparisons, std::vector<int>* Array, int left, int middle, int right)
462  {
463  assert(swaping != nullptr);
464  assert(comparisons != nullptr);
465  assert(swaping != comparisons);
466  assert(Array != nullptr);
467 
468  int i = middle + 1;
469  int j = left;
470 
471  std::vector<int>merged;
472 
473  while (i <= right && j <= middle)
474  {
475  if ((*Array)[i] <= (*Array)[j])
476  merged.push_back((*Array)[i++]);
477  else
478  merged.push_back((*Array)[j++]);
479 
480  (*comparisons)++;
481  (*swaping)++;
482  }
483 
484  while (j <= middle)
485  {
486  merged.push_back((*Array)[j++]);
487  (*swaping)++;
488  }
489 
490  while (i <= right)
491  {
492  merged.push_back((*Array)[i++]);
493  (*swaping)++;
494  }
495 
496  for (i = left; i <= right; i++)
497  {
498  (*Array)[i] = merged[i - left];
499  (*swaping)++;
500  }
