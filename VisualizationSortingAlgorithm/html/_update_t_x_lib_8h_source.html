248  void DrawFoneOfPushedButton(int x0, int y0, int x1, int y1, int R, COLORREF BackColor)
249  {
250  const int gap = MIN(x1 - x0, y1 - y0) / 2; // число эелементов градиента
251  const int r = R;
252 
253  R = 0;
254 
255  if ((x1 - x0) < (y1 - y0)) // определяем координаты начальной позиции градиента
256  {
257  y1 -= (x1 - x0) / 2;
258  y0 += (x1 - x0) / 2;
259  x0 = (x1 + x0) / 2;
260  x1 = x0;
261  }
262  else
263  {
264  x1 -= (y1 - y0) / 2;
265  x0 += (y1 - y0) / 2;
266  y0 = (y1 + y0) / 2;
267  y1 = y0;
268  }
269 
270  for (int i = 1; i <= gap; i++)
271  {
272  const COLORREF color = RGB(MAX(txExtractColor(BackColor, TX_RED) - 1.3 * i, 0), // получаем цвет для
273  MAX(txExtractColor(BackColor, TX_GREEN) - 1.3 * i, 0), // каждого элемента
274  txExtractColor(BackColor, TX_BLUE)); // градиента
275 
276  if (gap - i <= r) // если приблежаемся к границам кнопки делаем скругление, чтобы не было торчаших углов
277  R = r - (gap - i);
278 
279  txSetColor (color);
280  txSetFillColor (color);
281 
282  DrawGentleRectangle(x0, y0, x1, y1, R, color);
283 
284  //переходим к следующему элементу градиента
285  x0 -= 1;
286  y0 -= 1;
287  x1 += 1;
288  y1 += 1;
289  }
290  }
